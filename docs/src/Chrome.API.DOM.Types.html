                    , className           :: Maybe String
                    , value               :: Maybe Value
                    , unserializableValue :: Maybe String
                    , description         :: Maybe String
                    , objectId            :: Maybe String
                    } deriving Show


$(deriveJSON defaultOptions{
     omitNothingFields = True
   , fieldLabelModifier = let f "type" = "_type"
